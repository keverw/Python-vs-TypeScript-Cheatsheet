Python vs TypeScript Cheatsheet

1. Casing & Naming Conventions
Python (PEP 8)
Variables snake_case
Functions snake_case
Classes PascalCase
Constants UPPER_SNAKE_CASE
Private _internal_var
TypeScript
Variables camelCase
Functions camelCase
Classes PascalCase
Constants UPPER_SNAKE_CASE
Private private internalVar
2. Variables & Types
Python
# Dynamic typing
x = 10
name = "Alice"
is_active = True

# With Type Hints
count: int = 5
messages: list[str] = ["hello"]

# Primitive Types
# int, float, str, bool, None
# list, dict, set, tuple
TypeScript
// Static typing
let x: number = 10;
const name: string = "Alice";
let isActive: boolean = true;

// Type Inference
let count = 5;
const messages: string[] = ["hello"];

// Primitive Types
// number, string, boolean, null, undefined
// Array, object, Set, [T, T]
3. Data Structures
Python
# List (Array)
items = [1, 2, 3]
items.append(4)
first = items[0]

# Dictionary (Map/Object)
data = {"a": 1, "b": 2}
data["c"] = 3
val = data.get("a")
keys = data.keys()

# Set (Unique)
unique = {1, 2, 2, 3}
unique.add(4)
TypeScript
// Array
const items = [1, 2, 3];
items.push(4);
const first = items[0];

// Object (Dictionary)
const data: Record = { a: 1, b: 2 };
data["c"] = 3;
const val = data["a"];
const keys = Object.keys(data);

// Map (Ordered Key-Value)
const map = new Map([["a", 1]]);
map.set("b", 2);

// Set (Unique)
const unique = new Set([1, 2, 2, 3]);
unique.add(4);
4. Type Checking
Python
# Check type
type(x) == int
isinstance(x, int)

# Check inheritance
issubclass(Dog, Animal)
TypeScript
// Check primitive type
typeof x === 'number'

// Check class instance

5. Control Flow
Python
# If / Else
if x > 10:
    print("Big")
elif x == 10:
    print("Ten")
else:
    print("Small")

# Match (Switch)
match status:
    case 200: print("OK")
    case _: print("Unknown")

# Loops
for i in range(5):
    print(i)

for item in items:
    print(item)
TypeScript
// If / Else
if (x > 10) {
    console.log("Big");
} else if (x === 10) {
    console.log("Ten");
} else {
    console.log("Small");
}

// Switch
switch (status) {
    case 200: console.log("OK"); break;
    default: console.log("Unknown");
}

// Loops
for (let i = 0; i < 5; i++) {
    console.log(i);
}

for (const item of items) {
    console.log(item);
}
6. Functions
Python
def add(a: int, b: int) -> int:
    return a + b

# Default args
def greet(name="User"):
    print(f"Hello {name}")

# Lambda
square = lambda x: x * x
TypeScript
function add(a: number, b: number): number {
    return a + b;
}

// Default args
function greet(name: string = "User"): void {
    console.log(`Hello ${name}`);
}

// Arrow Function
const square = (x: number): number => x * x;
7. Classes
Python
class Dog:
    species = "Canis"

    def __init__(self, name: str):
        self.name = name
        self._age = 0

    def bark(self):
        print(f"{self.name} says Woof!")

    @property
    def age(self):
        return self._age
TypeScript
class Dog {
    static species = "Canis";
    public name: string;
    protected age: number = 0;

    constructor(name: string) {
        this.name = name;
    }

    bark(): void {
        console.log(`${this.name} says Woof!`);
    }

    get ageVal(): number {
        return this.age;
    }
}
8. Async / Await
Python
import asyncio

async def fetch_data():
    await asyncio.sleep(1)
    return "Data"

async def main():
    data = await fetch_data()
    print(data)

if __name__ == "__main__":
    asyncio.run(main())
TypeScript
// Returns a Promise
async function fetchData(): Promise {
    await new Promise(r => setTimeout(r, 1000));
    return "Data";
}

async function main() {
    const data = await fetchData();
    console.log(data);
}

main();

11. Error Handling
Python
try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"Error: {e}")
finally:
    print("Cleanup")

raise ValueError("Invalid")
TypeScript
try {
    throw new Error("Oops");
} catch (e) {
    console.log(e);
} finally {
    console.log("Cleanup");
}

throw new Error("Invalid");

13. String Manipulation
Python
name = "World"
msg = f"Hello {name}"

text = "Hello World"
# Slicing [Start:End:Step]
a = text[0:5]   # "Hello"
b = text[:5]    # "Hello"
c = text[6:]    # "World"
d = text[6:11]  # "World"
e = text[-1]    # "d"
f = text[::-1]  # Reverse
TypeScript
const name = "World";
const msg = `Hello ${name}`;

const str = "Hello World";
// Slice / Substring
const a = str.slice(0, 5);
const b = str.slice(0, 5);
const c = str.slice(6);
const d = str.slice(6, 11);
const e = str.slice(-1);
const f = str.split('').reverse().join('');
